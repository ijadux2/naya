func main(): int {
    player: Player = {
        position = Vector(float64){x = 400.0, y = 300.0},
        velocity = Vector(float64){x = 0.0, y = 0.0},
        color = raylib.MAROON,
        radius = 20.0
    }
    
    while !raylib.windowShouldClose() {
        player.update(raylib.getFrameTime())
        player.draw()
    }
    
    return 0
}
