Functions

Function Declaration

func function_name(param1: type1, param2: type2): return_type {
    // function body
    return value
}

Generic Functions

func max(T: type)(a: T, b: T): T {
    if a > b {
        return a
    }
    return b
}

Methods

Vector: type = struct(T: type) {
    x: T
    y: T
    
    func magnitude(self: Vector(T)): float64 {
        return sqrt(self.x * self.x + self.y * self.y)
    }
}

Heap Allocation

func main(): int {
    data: []int = alloc([]int, 100)  // Heap allocated
    defer free(data.ptr)  // Automatic cleanup
    
    // Use data...
    return 0
}

func main(): int {
    arena: Arena = {}
    defer arena.deinit()
    
    temp_data: []string = arena.alloc([]string, 50)
    // Use temp_data...
    
    return 0
}

Error Handling

Result Types

func safe_divide(a: int, b: int): Result[int, string] {
    if b == 0 {
        return Err("Division by zero")
    }
    return Ok(a / b)
}

Try-Catch

func main(): int {
    result: int = try safe_divide(10, 2) or return 1
    print("Result: {result}")
    return 0
}

Pattern Matching

match result {
    Ok(value) => print("Success: {value}"),
    Err(error) => print("Error: {error}")
}
