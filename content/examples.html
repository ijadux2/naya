File Operations

Medium

Implement a simple cat utility using system calls.

System Calls File I/O Error Handling
cat.naya 
func main(): int {
    buffer: [4096]uint8
    
    while true {
        bytes_read: uint = syscall.read(0, buffer, 4096)
        if bytes_read == 0 {
            break  // EOF
        }
        
        bytes_written: uint = syscall.write(1, buffer, bytes_read)
        if bytes_written != bytes_read {
            print("Write error")
            return 1
        }
    }
    
    return 0
}

Process Management

Medium

Create a simple process monitor using system calls.

Process Control System Info Memory
process_monitor.naya 
import sys

func get_process_info(pid: int): Result[ProcessInfo, string] {
    // Read process information from /proc filesystem
    stat_file: string = "/proc/{pid}/stat"
    file: int = sys.open(stat_file, sys.O_RDONLY)
    
    if file < 0 {
        return Err("Cannot open process file")
    }
    
    defer sys.close(file)
    
    buffer: [1024]uint8
    bytes_read: uint = sys.read(file, buffer, 1024)
    
    if bytes_read == 0 {
        return Err("Cannot read process file")
    }
    
    // Parse process information (simplified)
    return Ok(parse_process_info(buffer))
}

func main(): int {
    current_pid: int = sys.getpid()
    
    match get_process_info(current_pid) {
        Ok(info) => {
            print("Process ID: {info.pid}")
            print("Parent PID: {info.ppid}")
            print("Command: {info.comm}")
        },
        Err(error) => {
            print("Error: {error}")
            return 1
        }
    }
    
    return 0
}

Raylib Integration

Medium

Simple graphics application using Raylib library.

Graphics Game Loop C ABI
raylib_game.naya 
import raylib

Player: type = struct {
    x: float64
    y: float64
    radius: float64
    color: uint32
}

func main(): int {
    raylib.initWindow(800, 600, "Naya + Raylib")
    defer raylib.closeWindow()
    
    player: Player = {
        x = 400.0,
        y = 300.0,
        radius = 20.0,
        color = raylib.MAROON
    }
    
    while !raylib.windowShouldClose() {
        raylib.beginDrawing()
        raylib.clearBackground(raylib.RAYWHITE)
        
        // Update player
        if raylib.isKeyDown(raylib.KEY_RIGHT) {
            player.x = player.x + 5.0
        }
        if raylib.isKeyDown(raylib.KEY_LEFT) {
            player.x = player.x - 5.0
        }
        if raylib.isKeyDown(raylib.KEY_UP) {
            player.y = player.y - 5.0
        }
        if raylib.isKeyDown(raylib.KEY_DOWN) {
            player.y = player.y + 5.0
        }
        
        // Keep player in bounds
        if player.x < player.radius {
            player.x = player.radius
        }
        if player.x > 800 - player.radius {
            player.x = 800 - player.radius
        }
        if player.y < player.radius {
            player.y = player.radius
        }
        if player.y > 600 - player.radius {
            player.y = 600 - player.radius
        }
        
        // Draw player
        raylib.drawCircleV(
            raylib.Vector2{x = player.x, y = player.y},
            player.radius,
            player.color
        )
        
        raylib.endDrawing()
    }
    
    return 0
}

Generic Data Structures

Hard

Implement generic data structures with compile-time evaluation.

Generics Comptime Memory Management
generics.naya 
Vector: type = struct(T: type) {
    data: ptr[T]
    len: uint
    cap: uint
}

func Vector(T: type).new(capacity: uint): Vector(T) {
    data: ptr[T] = alloc(T, capacity)
    return Vector(T){
        data = data,
        len = 0,
        cap = capacity
    }
}

func Vector(T: type).append(self: Vector(T), item: T): void {
    if self.len >= self.cap {
        // Resize logic would go here
        return
    }
    
    self.data[self.len] = item
    self.len = self.len + 1
}

func Vector(T: type).get(self: Vector(T), index: uint): T {
    if index >= self.len {
        panic("Index out of bounds")
    }
    return self.data[index]
}

func main(): int {
    // Create vector of integers
    numbers: Vector(int) = Vector(int).new(10)
    
    // Add elements
    numbers.append(10)
    numbers.append(20)
    numbers.append(30)
    
    // Access elements
    for i in 0..numbers.len {
        print("Number {i}: {numbers.get(i)}")
    }
    
    // Create vector of strings
    strings: Vector(string) = Vector(string).new(5)
    strings.append("Hello")
    strings.append("Naya")
    strings.append("World")
    
    return 0
}
